57 */
58 template<class CoordinateType>
59 std::vector<CoordinateType> Triangle<CoordinateType>::calculateBarycentricCoordinates(const Point<CoordinateType>& p) const{
60  std::vector<CoordinateType> barCoordinates;
61  Point<CoordinateType> pa = p - a;
62  Point<CoordinateType> pb = p - b;
63  Point<CoordinateType> pc = p - c;
64  barCoordinates.push_back((pb.getX() * pc.getY()) - (pb.getY() * pc.getX()));
65  barCoordinates.push_back((pc.getX() * pa.getY()) - (pc.getY() * pa.getX()));
66  barCoordinates.push_back((pa.getX() * pb.getY()) - (pa.getY() * pb.getX()));
67  return barCoordinates;
68 }

146 /* Calculate the distance between a triangle and another shape.
147 * The distance between two triangles is the shortest distance between two points from each shape
148 * points represents the couple of points, to get the translation to perform
149 */
150 template<class CoordinateType>
151 double Triangle<CoordinateType>::distance(Shape<CoordinateType>* shape, std::vector<Point<CoordinateType>>& points) const {
152  // First, get the points from each triangle
153  std::vector<Point<CoordinateType>> selectedPiecePoints = this->getPoints();
154  std::vector<Point<CoordinateType>> otherPoints = shape->getPoints();
155  double minDist = DBL_MAX;
156  double dist;
157  // Calculate the distance between each possible couple of points (p1, p2), where p1 is a point of shape and p2 is a point of the current instance
158  for(auto& point : selectedPiecePoints) {
159  for(auto& otherPoint : otherPoints) {
160  dist = point.distance(otherPoint);
161  if(dist < minDist) {
162  points[0] = point;
163  points[1] = otherPoint;
164  minDist = dist;
165  }
166  }
167  }
168  return minDist;
169 }
